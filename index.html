1. Sum of all elements in an array

12. Remove duplicate elements from array
#include <stdio.h>
int main() {
    int n, arr[100];
    printf("Size: "); scanf("%d", &n);
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n; i++) {
        for(int j=i+1; j<n; j++) {
            if(arr[i] == arr[j]) {
                for(int k=j; k<n-1; k++) arr[k] = arr[k+1];
                n--; j--;
            }
        }
    }
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    return 0;
}

    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {

}
20. Binary search
#include <stdio.h>
int main() {
    int n, arr[100], target, low, high, mid, found = 0;
    printf("Size (Sorted): "); scanf("%d", &n);
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    printf("Target: "); scanf("%d", &target);
    low = 0; high = n-1;
    while(low <= high) {
        mid = (low + high) / 2;
        if(arr[mid] == target) { printf("Found at index %d\n", mid); found = 1; break; }
        else if(arr[mid] < target) low = mid + 1;
        else high = mid - 1;
    }
    if(!found) printf("Not Found\n");
    return 0;
}

25. Count vowels, consonants, digits, special ch
#include <stdio.h>
int main() {
    char str[100]; int v=0, c=0, d=0, s=0;
    printf("String: "); scanf("%s", str);
    for(int i=0; str[i]!='\0'; i++) {
        char ch = str[i];
        if((ch>='a'&&ch<='z') || (ch>='A'&&ch<='Z')) {
            if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'||ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U') v++;
            else c++;
        }
        else if(ch>='0' && ch<='9') d++;
        else s++;
    }
    printf("Vowels: %d\nConsonants: %d\nDigits: %d\nSpecial: %d\n", v, c, d, s);
    return 0;
}
